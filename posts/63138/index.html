# Macos 系统

复制下面的保存为.sh 格式，用的时候拖放到终端里面运行就行
#!/bin/bash echo "正在打开新的微信！！！" nohup /Applications/WeChat.app/Contents/MacOS/WeChat > /dev/null 2>&1 &

# Windows 系统

桌面新建一个文本文档，多开几个复制几行，另存为 bat 文件
start "" "C:\Program Files (x86)\Tencent\WeChat\WeChat.exe"
